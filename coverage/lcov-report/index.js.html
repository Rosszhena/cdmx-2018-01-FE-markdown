const marked = require('marked');
 
// Funcion que valida la ruta si es relativa o absoluta
const validateLink = (route) =>{
  if (path.isAbsolute(route) != true) {
    let fileAbsolut = path.resolve(route);
    return fileAbsolut;
  } else return route; 
};
